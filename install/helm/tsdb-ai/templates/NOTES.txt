1. Check everything is running:

     kubectl -n {{ include "tsdb-ai.namespace" . }} get pods,svc,ingress,pvc

2. Wait for readiness:

     kubectl -n {{ include "tsdb-ai.namespace" . }} rollout status deploy/tsdb-server
{{- if (index .Values "tsdb-ui" "enabled") }}
     kubectl -n {{ include "tsdb-ai.namespace" . }} rollout status deploy/tsdb-ui
{{- end }}

3. Reach the stack:
{{- if .Values.ingress.enabled }}

   Ingress is enabled. Point your DNS (or /etc/hosts for testing) at the
   nginx ingress controller's external IP:

     kubectl -n ingress-nginx get svc ingress-nginx-controller \
       -o jsonpath='{.status.loadBalancer.ingress[0].ip}'

   Then open:

     http{{ if .Values.ingress.tls.enabled }}s{{ end }}://{{ .Values.ingress.host }}/

   Ingress route map:
     /                    → tsdb-ui:3000
     /api, /internal      → tsdb-server:8080
     /ingest_samples      → tsdb-server:8080  (Prometheus-compatible write)
     /forecast, /patterns → tsdb-server:8080  (AI endpoints)
     /causal, /relationships
     /qgw                 → tsdb-server:8081  (PromQL reads)
     /vectors, /search    → tsdb-server:8084  (vector store)
     /sse                 → tsdb-server:8000  (MCP / Claude)
{{- else }}

   Ingress is disabled. Port-forward to reach each service:

     kubectl -n {{ include "tsdb-ai.namespace" . }} port-forward svc/tsdb-ui 3000:3000
     kubectl -n {{ include "tsdb-ai.namespace" . }} port-forward svc/tsdb-server 8080:8080
     open http://localhost:3000
{{- end }}

4. Tail logs if something looks off:

     kubectl -n {{ include "tsdb-ai.namespace" . }} logs -f -l app=tsdb-server
{{- if (index .Values "tsdb-scraper" "enabled") }}
     kubectl -n {{ include "tsdb-ai.namespace" . }} logs -f -l app=tsdb-scraper
{{- end }}
